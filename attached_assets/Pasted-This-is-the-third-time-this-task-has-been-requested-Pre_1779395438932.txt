const keyFromRequest = (req: any) =>
  (req.headers['x-forwarded-for'] as string)
    ?.split(',')[0]?.trim() || req.ip || 'unknown';

const jsonHandler = (msg: string) => (req: any, res: any) => {
  const retryAfter = Math.ceil(
    (req.rateLimit.resetTime.getTime() - Date.now()) / 1000
  );
  res.status(429).json({
    error: 'rate_limited',
    message: msg,
    retryAfter,
  });
};

// EXPENSIVE — question generation (~8000 tokens per call)
const expensiveLimiter = rateLimit({
  windowMs: 60 * 60 * 1000,
  max: 5,
  keyGenerator: keyFromRequest,
  standardHeaders: true,
  legacyHeaders: false,
  handler: jsonHandler(
    'Question generation limit reached. Please wait before generating more questions.'
  ),
});

// STANDARD — explanations, study tips, analysis (~750-1500 tokens)
const standardLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 30,
  keyGenerator: keyFromRequest,
  standardHeaders: true,
  legacyHeaders: false,
  handler: jsonHandler(
    'Too many requests. Please slow down.'
  ),
});

// LIGHT — tutor, NHS prep, adaptive chat
const lightLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 60,
  keyGenerator: keyFromRequest,
  standardHeaders: true,
  legacyHeaders: false,
  handler: jsonHandler(
    'Too many messages. Please wait a moment.'
  ),
});
